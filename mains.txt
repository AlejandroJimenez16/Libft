#ft_striteri

void to_upper(unsigned int index, char *c)
{
    if (index % 2 == 0)
        *c = ft_toupper(*c);
}

#include <stdio.h>

int main()
{
    char str[] = "hola mundo";

    printf("Antes: %s\n", str);

    ft_striteri(str, to_upper);

    printf("Después: %s\n", str);

}
