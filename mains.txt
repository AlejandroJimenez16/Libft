#ft_memset
-----------

#include <string.h>
#include <stdio.h>

int main(void)
{
    char buffer[10];
    size_t i;

    ft_memset(buffer, 'c', sizeof(buffer));

    i = 0;
    while(i < sizeof(buffer))
    {
        printf("%c", buffer[i]);
        i++;
    }
}
